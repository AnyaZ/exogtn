open /portal/public/classic/
echo -CategoryManagement-
windowMaximize
click link=Sign in
waitForElementPresent username
type username root
type password gtn
clickAndWait xpath=//div[@id='UIPortalLoginFormAction']/div/div/div
echo --Select "Application Registry"
waitForElementPresent link=Application Registry
click link=Application Registry
echo --Add Category
waitForElementPresent xpath=//div[@id='UIApplicationOrganizer']//div[@class='UIControlbar']/div[1]
click xpath=//div[@id='UIApplicationOrganizer']//div[@class='UIControlbar']/div[1]
waitForElementPresent name
type name test_name_category_10
type displayName test_displayname_category_10
type description test_description_category_10
echo --Select permissions
click xpath=//div[@class='WorkingArea']//div[@class='TabsContainer']/div[2]//div[@class='MiddleTab']
waitForElementPresent link=Add Permission
click link=Add Permission
waitForElementPresent link=Platform
click link=Platform
waitForElementPresent link=Administrators
click link=Administrators
waitForElementPresent link=manager
click link=manager
waitForText //div[@id='PermissionGrid']/table/tbody/tr/td[2]/div manager
click link=Save
waitForElementPresent xpath=//div[@id='UIApplicationOrganizer']//a[@class='ControlIcon EditIcon']
waitForTextPresent test_displayname_category_10
verifyTextPresent test_displayname_category_10
echo --Edit Category
click xpath=//div[@id='UIApplicationOrganizer']//a[@class='ControlIcon EditIcon']
waitForElementPresent displayName
type displayName test_displayname_edit_10
waitForElementPresent link=Save
click link=Save
waitForTextPresent test_displayname_edit_10
verifyTextPresent test_displayname_edit_10
echo --Delete Category
click xpath=//div[@class='SelectedTab']//a[@class='ControlIcon DeleteIcon']
waitForConfirmation Are you sure to delete this category and all applications on it?
waitForTextNotPresent test_displayname_edit_10
verifyTextNotPresent test_displayname_edit_10
clickAndWait //div[3]/a
