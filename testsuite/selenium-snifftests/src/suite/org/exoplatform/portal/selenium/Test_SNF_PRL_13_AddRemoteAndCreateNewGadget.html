Test_SNF_PRL_13_AddRemoteAndCreateNewGadget
open /portal/public/classic/
echo -AddGadget-
windowMaximize
click link=Sign in
waitForElementPresent username
type username root
type password gtn
click xpath=//div[@id='UIPortalLoginFormAction']/div/div/div
echo --Go to Application registry--
waitForElementPresent link=Application Registry
click link=Application Registry
waitForElementPresent link=Gadget
click link=Gadget
echo --Add a remote gadget--
waitForElementPresent xpath=//div[@class='IconControl AddNewIcon']
click xpath=//div[@class='IconControl AddNewIcon']
waitForElementPresent url
type url http://www.google.com/ig/modules/datetime.xml
waitForElementPresent link=Add
click link=Add
waitForTextPresent Gadget Details
verifyTextPresent Gadget Details
echo --Delete remote gadget--
click //div[@class='SelectedItem ItemContent']/a[2]
waitForConfirmation Are you sure to delete this gadget?
waitForElementNotPresent //a[@title='Date & Time']
verifyElementNotPresent //a[@title='Date & Time']
echo --Add a new gadget--
waitForElementPresent //div[@id='UIApplicationRegistryPortlet']/div[2]/div[2]/div/div/div[2]
click //div[@id='UIApplicationRegistryPortlet']/div[2]/div[2]/div/div/div[2]
waitForElementPresent name
type name SNF_PRL_13
waitForElementPresent link=Save
click link=Save
waitForTextPresent Gadget Details
waitForElementPresent //div[@class='SelectedItem ItemContent']//a[2]
click //div[@class='SelectedItem ItemContent']//a[2]
waitForConfirmation Are you sure to delete this gadget?
waitForTextNotPresent link=hello world example
verifyTextNotPresent link=hello world example
clickAndWait link=Sign out
